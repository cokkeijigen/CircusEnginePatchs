
file(GLOB_RECURSE SOURCES "${CMAKE_CURRENT_LIST_DIR}/*.cpp")
add_library(Utillibs OBJECT ${SOURCES})

set(DETOURS_DIR   "${CMAKE_CURRENT_LIST_DIR}/detours")
set(YY_THUNKS_DIR "${CMAKE_CURRENT_LIST_DIR}/yy-thunks")

file(GLOB HEADER_DIR ALL_FILES_AND_DIRS 
    "${CMAKE_CURRENT_LIST_DIR}"
    "${CMAKE_CURRENT_LIST_DIR}/*"
    "${CMAKE_CURRENT_LIST_DIR}/*/*"
)

target_include_directories(Utillibs PUBLIC  
    ${DETOURS_DIR} 
    ${HEADER_DIR}
)

target_link_libraries(Utillibs 
    "${DETOURS_DIR}/detours.lib"
    "${YY_THUNKS_DIR}/YY_Thunks_for_Win2K.obj"
)
